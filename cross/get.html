            function push(task) {
                !isDuplicate(task) && stack.push(task);
            }

            function unShift(task) {
                stack.unshift(task);
            }

            function isDuplicate(task) {
                var i = 0, len = stack.length;
                for (; i < stack.length; i++) {
                    if (stack[i] === task) {
                        return true;
                    }
                }
                return false;
            }

            function remove(task) {
                var i = 0, len = stack.length;
                for (; i < len; i++) {
                    if (stack[i] === task) {
                        delete stack[i];
                        stack.splice(i, 1);
                    }
                }
                return false;
            }

            function run(data) {
                stack.length && (stack.shift()(data) !== false) && run();
            }

            return {
                run: run,
                push: push,
                unShift: unShift,
                remove: remove,
                stack: stack
            }
        }
